To bootstrap a Clojure Project:
#+begin_src bash :noeval
  nix flake init -t github:chrischambers/nix#clojure
#+end_src

To bootstrap a Python Project:
#+begin_src bash :noeval
  nix flake init -t github:chrischambers/nix#python
#+end_src
